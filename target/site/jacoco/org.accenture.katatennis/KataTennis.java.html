package org.accenture.katatennis;

import org.accenture.model.*;

public class KataTennis {

	public static void main(String[] args) throws Exception {
		
		Player playerOne = new Player("Reda");
		Player playerTwo = new Player("Rafael");
		Game gameExample = new Game(playerOne, playerTwo);
		
		//15-0
		gameExample.playerOneWinsBall();
		//15-15
		gameExample.playerTwoWinsBall();
		//15-30
		gameExample.playerTwoWinsBall();
		//30-30
		gameExample.playerOneWinsBall();
		//40-30
		gameExample.playerOneWinsBall();
		//40-40 Deuce
		gameExample.playerTwoWinsBall();
		//40-40 Advantage player Two
		gameExample.playerTwoWinsBall();
		//40-40 Deuce
		gameExample.playerOneWinsBall();
		//40-40 Advantage player One
		gameExample.playerOneWinsBall();
		//40-40 Player One wins
		gameExample.playerOneWinsBall();
		//40-40 Player One has already won
		gameExample.playerOneWinsBall();
	}
}
